# Exemplo de execução do worker a cada 5 minutos
# /etc/systemd/system/autotrader-saida-worker.service
[Unit]
Description=AUTOTRADER-SAIDA Worker (monitor)

[Service]
Type=oneshot
User=roteiro_ds
WorkingDirectory=/home/roteiro_ds/AUTOTRADER-SAIDA
Environment=DATA_DIR=/home/roteiro_ds/AUTOTRADER-SAIDA/data
ExecStart=/usr/bin/python3 /home/roteiro_ds/AUTOTRADER-SAIDA/worker_saida_v2.py

# /etc/systemd/system/autotrader-saida-worker.timer
[Unit]
Description=AUTOTRADER-SAIDA Worker Timer (5 min)

[Timer]
OnBootSec=60
OnUnitActiveSec=300
AccuracySec=30

[Install]
WantedBy=timers.target
